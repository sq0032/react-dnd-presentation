var file_source = {
    beginDrag: function(props){
        //Do something when drag begins
        //Usually return component props to monitor
        return {file: props.file};
    },
    canDrag: function(props, monitor){
        //Return true or false to tell DnD if the component is draggable
        return true;
    },
    endDrag: function(props, monitor){
        //Do something when drag ends
        //The action or Ajax usually get called here
    }
};

    };
};

Put everything together!

Use connect to wrap your component


var file_source = {
  //overwrite DnD apis...
};

var file_source_collect = function(connect, monitor){
  return {
    connectDragSource: connect.dragSource()
    //monitor attributes...
  };
};

@DragSource('FILE', file_source, file_source_collect)
class FileItem extends Component{
  //... 
  render(){
    const { connectDragSource } = this.props;
    return connectDragSource(
      //your component jsx
    );
  }
}
